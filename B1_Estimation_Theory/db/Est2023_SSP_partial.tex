The random variables $S$ and $X$ are related through the likelihood function
\begin{equation*}
p_{X|S}(x|s) = \frac16 x ^2 s^3 \exp(-sx),    \qquad s \ge 0, \quad x \ge 0
% p_{X|S}(x|s) = x s^2 \exp(-sx),    \qquad s \ge 0, \quad x \ge 0
\end{equation*}
The prior distribution of $S$ is given by
\begin{equation*}
p_{S}(s) = \exp(-s), \qquad  s \ge 0
\end{equation*}

\begin{parts}
\part Compute the MAP estimate of $S$ given $X$, $\hat{S}_{\text{MAP}}$
\part Compute the ML estimate of $S$ given a collection of IID observations $\{x_k, k=0, 	\ldots K-1\}$ from $X$.
\end{parts}

%     = \arg \max_{s\ge 0} \left(2 \log(s) -(s+1)x \right)   \\

\begin{align*}
\hat{S}_\text{MAP} 
    &= \arg \max_s p_{S|X}(s | x) 
     = \arg \max_s p_{X|S}(x | s) p_S(s)   \\
    &= \arg \max_{s\ge 0} \left(x^2 s^3 \exp(-(x+1)s)  \right)
     = \arg \max_{s\ge 0} \left(3 \log(s) -(x+1)s \right)   \\
    &= \frac3{x+1}
\end{align*}

%\end{align*}
\begin{align*}
\hat{S}_\text{ML} 
    &= \arg \max_s \sum_{k=0}^{K-1} \log\left(p_{X|S}(x_k | s) \right)
     = \arg \max_s \left(3K \log(s) - s \sum_{k=0}^{K-1} x_k \right)   \\
    &= \frac{3K}{\sum_{k=0}^{K-1} x_k}
\end{align*}
