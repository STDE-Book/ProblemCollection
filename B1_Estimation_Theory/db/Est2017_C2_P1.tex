\question[20] % VGV	

La distribución conjunta de dos variables aleatorias $S$ y $X$ está dada por:
\[
p_{S,X}(s,x) = 12 s^2 , \qquad 0 < s < x, \qquad 0 < x < 1  
\]
\begin{parts}
\part Obtenga el estimador de $S$, $\hat{S}_{\text{c}}$, que mínimiza el riesgo conditional de la siguiente función de coste:
\[
c(s, \hat{s} ) = \frac{\left(s - \hat{s} \right)^2 } {s} 
\]
\part Determine el coste medio condicionado a la observación, $\EE\{c(S, \hat{S}_{\text{c}} ) |X=x\}$ para el estimador obtenido en el apartado a).
\end{parts}

\begin{solution}
\begin{parts}
\part $\hat{S}_{\text{c}} = \frac{2 X}{3} $ 
\part  $\EE\{c(S, \hat{S}_{\text{c}} ) |X=x\} = \frac{x}{12}$
\end{parts}
\end{solution}
