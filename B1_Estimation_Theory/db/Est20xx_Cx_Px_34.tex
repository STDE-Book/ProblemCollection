\question 
 Se desea estimar la v.a. $S$ a partir de la observación $X$ dada por:

$$X=S+N_1+N_2$$

donde $S$ es una v.a. gaussiana de media $m_s$ y varianza $v_s$, y donde $N_1$ y $N_2$ son dos variables de ruido, independientes de $S$, cuya d.d.p. conjunta viene dada por:
$$
p_{N_1,N_2}\left(n_1,n_2\right) 
	\sim G\left ( \begin{bmatrix} 0 \\ 0 \end{bmatrix},
	              \begin{bmatrix} v_1 & v_{12} \\ v_{12} & v_2 \end{bmatrix}
	      \right)
$$

Obtenga:
\begin{parts}
\part El estimador de mínimo error cuadrático medio de $S$ a la vista de $X$, $\widehat S_{\text{MMSE}}$.
\part El estimador de máximo a posteriori de $S$ a la vista de $X$, $\widehat S_{\text{MAP}}$.
\part Calcule el sesgo y varianza de ambos estimadores. 
\end{parts}

\begin{solution}
\begin{parts}
\part  
\begin{align*}
\SMSE = m_S + \frac{v_{SX}}{v_X} (X - m_X)
\end{align*}
donde 
\begin{align*}
m_X &= \EE\{X\} = \EE\{S + N_1 + N_2\} = \EE\{S\}  \\
v_{SX} &= \EE\{(S-m_S)(X-m_X)\} 
        = \EE\left\{(S - m_S) \left(S - m_S + N_1 + N_2\right) \right\}             \\
       &= \EE\left\{(S - m_S)^2 \right\} + \EE\left\{(S - m_S) (N_1 + N_2) \right\}  \\
       &= v_S + \EE\left\{(S - m_S)\right\} \EE\left\{(N_1 + N_2) \right\}             \\
       &= v_S   \\
v_{X} &= \EE\{(X-m_X)^2\} 
        = \EE\left\{(S - m_S + N_1 + N_2)^2 \right\}                                   \\
       &= \EE\left\{(S - m_S)^2 \right\} + 2 \EE\left\{(S - m_S) (N_1 + N_2) \right\}
                                         + \EE\left\{(N_1 + N_2)^2 \right\}                       \\
       &= v_S + \EE\left\{(S - m_S)\right\} \EE\left\{(N_1 + N_2) \right\}   \\
       &= v_S + v_1 + v_2 + 2v_{12}
\end{align*}
Por tanto
\begin{align*}
\SMSE = m_S + \frac{v_S}{v_S + v_1 + v_2 + 2v_{12}} (X - m_S)   
\end{align*}

\part $\SMAP = \SMSE$

\part $\EE\left\{\SMSE - S \right\} = 
       \EE\left\{\SMAP - S \right\} = 0$
\begin{align*}
{\rm Var}\left\{ \SMAP \right\} 
	 = {\rm Var} \left\{ \SMSE \right\} 
	&= \EE\left\{ \left(\SMSE-m_S\right)^2 \right\}   \\
	&= \EE\left\{ \left(\frac{v_S}{v_S + v_1 + v_2 + 2v_{12}} (X - m_S)\right)^2 \right\}   \\
	&= \frac{v_S^2}{(v_S + v_1 + v_2 + 2v_{12})^2} (v_S + v_1 + v_2 + 2v_{12})   \\
    &= \frac{v_S^2}{v_S + v_1 + v_2 + 2 v_{12}}
\end{align*}                                   
\end{parts}
\end{solution}

\question 
We wish to estimate a random variable $S$ from the observation of another random variable $X$ given by:
$$X=S+N_1+N_2$$
where $S$ is Gaussian-distributed, with mean and variance $m_s$ and $v_s$, respectively, and where $N_1$ and $N_2$ are two noise random variables, independent of $S$, and with joint p.d.f. 
$$
p_{N_1,N_2}\left(n_1,n_2\right) 
	\sim G\left ( \begin{bmatrix} 0 \\ 0 \end{bmatrix},
	              \begin{bmatrix} v_1 & v_{12} \\ v_{12} & v_2 \end{bmatrix}
	      \right)
$$

Obtain:
\begin{parts}
\part The minimum mean square error estimator of $S$ given $X$, $\widehat S_{\text{MMSE}}$.
\part The maximum a posteriori estimator of $S$ given $X$, $\widehat S_{\text{MAP}}$.
\part The bias and variance of both estimators. 
\end{parts}

\begin{solution}
\begin{parts}
\part  
\begin{align*}
\SMSE = m_S + \frac{v_{SX}}{v_X} (X - m_X)
\end{align*}
where 
\begin{align*}

       &= v_S + \EE\left\{(S - m_S)\right\} \EE\left\{(N_1 + N_2) \right\}             \\
       &= v_S   \\
v_{X} &= \EE\{(X-m_X)^2\}
