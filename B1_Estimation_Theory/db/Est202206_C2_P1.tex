\question[25] 

\ifspanish

\else

Random variables $S$ and $X$ follow the joint distribution
\begin{align*}
p_{S,X}(s, x) = 6 e^{2x}(e^{-x} -s) s,  \qquad\qquad  0 \le s \le e^{-x}, \qquad x \ge 0
\end{align*}


\begin{parts}
\part[7] Compute the maximum a posteriori estimate of $S$ given $X$, $\hat{S}_\text{MAP}$.
\part[8] Compute the minimum MSE estimate of $S$ given $X$, $\hat{S}_\text{MSE}$.
\part[8] Compute the Bayesian estimate $\hat{S}^*$ minimizing the risk given by the cost
\begin{align*}
c(S, \hat{S}) = \frac{(S-\hat{S})^2}{S}
\end{align*}
\end{parts}
\begin{solution}

\begin{parts}
\part The MAP estimate is
\begin{align*}
\hat{S}_\text{MAP}
    &= \argmax_s p_{S|X}(s| x)
     = \argmax_s \frac{p_{S,X}(s, x)}{p_X(x)}
     = \argmax_s p_{S,X}(s, x)   \\
    &= \argmax_s \{(e^{-x} -s) s \} 
     = \frac{1}{2} e^{-x},  \qquad x \ge 0
\end{align*}

\part The marginal distribution is
\begin{align*}
p_X(x) 
	&= \int_{-\infty}^{\infty} p_{S,X}(s,x) ds     \\
	&= 6 e^{2x} \int_{0}^{e^{-x}} (e^{-x} -s) s ds   
	 = 6 e^{2x} \left[\frac{e^{-x}}{2}s^2  - \frac{1}{3} s^3 \right]_{0}^{e^{-x}}     \\	
	&= e^{-x},  \qquad x \ge 0
\end{align*}
Therefore, the posterior distribution is
\begin{align*}
p_{S|X}(s|x) = 6 e^{3x}(e^{-x} -s) s,  \qquad\qquad  0 \le s \le e^{-x}, \qquad x \ge 0
\end{align*}
and the minimum MSE estimate is
\begin{align*}
\hat{s}_\text{MSE} 
	&= \mathbb{E}\{S|x\} 
 	 = \int_{-\infty}^{\infty} s p_{S|X}(s|x) ds     \\
 	&= 6 e^{2x} \int_{0}^{e^{-x}} (e^{-x} -s) s^2 ds     
	 = 6 e^{2x} \left[\frac{e^{-x}}{3}s^3  - \frac{1}{4} s^4 \right]_{0}^{e^{-x}}     \\	
	&= \frac{1}{2} e^{-x},  \qquad x \ge 0
\end{align*}

\part
The conditional risk is given by
\begin{align*}
\mathbb{E}\{c(S, \hat{s}) \mid x \} 
	&= \mathbb{E} \left\{ \frac{(S - \hat{s})^2}{S} \mid x\right\}    \\
	&= \mathbb{E} \left\{S - 2 \hat{s} + \frac{\hat{s}^2}{S} \mid x \right\}    \\
	&= \mathbb{E} \left\{S \mid x \right\} 
	 - 2 \hat{s} + \hat{s}^2 \mathbb{E} \left\{\frac{1}{S} \mid x \right\} 
\end{align*}
which is minimized, with respect to $\hat{s}$, for 
\begin{align*}
\hat{s}^* = \frac{1}{\mathbb{E} \left\{\frac{1}{S} | x \right\} }   \\
\end{align*}
For the given distribution
\begin{align*}
\mathbb{E} \left\{\frac{1}{S} \mid x \right\}
	&= \int_{-\infty}^{\infty} \frac{1}{s} p_{S|X}(s|x) ds    \\
 	&= 6 e^{3x} \int_{0}^{e^{-x}} (e^{-x} -s) ds 	
	 = 6 e^{3x} \left[e^{-x} s  - \frac{1}{2} s^2 \right]_{0}^{e^{-x}}     \\	
	&= 3 e^{x},  \qquad x \ge 0
\end{align*}
Therefore,
\begin{align*}
\hat{s}^* = \frac{1}{3} e^{-x},  \qquad x \ge 0
\end{align*}
\end{parts}

\end{solution}
