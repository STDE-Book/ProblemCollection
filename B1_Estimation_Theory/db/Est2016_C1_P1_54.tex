\question[20] % JCS

La variable aleatoria $X$ sigue la siguiente función de densidad de probabilidad:
\[
p_{X}(x) = a^2 x \exp\left(-a x \right) , \qquad x\ge 0
\]
siendo $a$ un parámetro desconocido.
\begin{parts}
\part Determine la expresión del estimador ML de $a$ a la vista de un conjunto $\mathcal{C} = \{(x^{(k)}),k=1,\ldots,K\}$ de $K$ realizaciones independientes de $X$.
\part Dado el conjunto de observaciones $\mathcal{C} = \{0.2, 0.5, 0.8, 1\}$, determine el valor de $\widehat A_{\text{ML}}$.% y obtenga el valor de la verosimilitud de $A_{\text{ML}}$ sobre dicho conjunto de observaciones.
\end{parts}

\begin{solution}
\begin{parts}
\part $\widehat A_{\text{ML}} = \dfrac{2K}{\sum_{k=1}^K x^{(k)}}$ 
\part $\widehat A_{\text{ML}} = 3.2 $%, $p_{\bf X}({\bf x}|a_{\text{ML}}) = 0.295$
\end{parts}
\end{solution}

\end{parts}

\begin{solution}
\begin{parts}
\part $\widehat A_{\text{ML}} = \dfrac{2K}{\sum_{k=1}^K x^{(k)}}$ 
\part $\widehat A_{\text{ML}} = 3.2 $%, $p_{\bf X}({\bf x}|a_{\text{ML}}) = 0.295$

\end{parts}
\end{solution}
