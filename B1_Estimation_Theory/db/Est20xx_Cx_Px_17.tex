\ifspanish

\question Se estima la varianza $v$ de una v.a. $X$ con media nula a partir de $K$ observaciones $\left\lbrace X_k \right\rbrace_{k=0}^{K-1} $  de dicha variable tomadas independientemente, utilizando el estimador 
 $$\hat{V}=\frac{1}{K}\left[\sum_{k=0}^{K-1} X_k\right]^2$$ 
\begin{parts}
\part   Determínese el sesgo del estimador.
\part   Para  $K=2$  y suponiendo  $\mathbb E\left\lbrace X^4 \right\rbrace=\alpha$, determínese la varianza del estimador.
  \end{parts}

\begin{solution}
\begin{parts}
\part 	Es insesgado		
\part  $\text{Var}\left\lbrace \hat{V} \right\rbrace=\displaystyle\frac{1}{2}\left[ \alpha + v^2\right] $
  \end{parts}
  \end{solution}

\question The variance $v$ of a zero-mean r.v. $X$ is estimated from $K$ independent observations of the variable, $\left\{ X_k \right\}_{k=0}^{K-1}$, using the following estimator:
$$\hat{V}=\frac{1}{K}\left[\sum_{k=0}^{K-1} X_k\right]^2$$ 
\begin{parts}
\part Find the bias of such an estimator.
\part For $K=2$, and assuming that $\EE\left\{ X^4 \right\}=\alpha$ is known, find the variance of the estimator.
  \end{parts}

\begin{solution}
\begin{parts}
\part The estimator is unbiased.
\part  $\text{var}\left\{\hat{V} \right\}= \dfrac{1}{2}\left[ \alpha + v^2\right] $
  \end{parts}
\end{solution}
