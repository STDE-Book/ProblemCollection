\question[20] % JCS

La distribución conjunta de dos variables aleatorias $S$ y $X$ está dada por:
\[
p_{S,X}(s,x) = \sqrt{\frac{x(1-x)}{2\pi}} \exp\left(-\frac{(s-x(1-x))^2}{2x(1-x)} \right) , \qquad 0 < x < 1, \qquad s \in \mathbb{R}  
\]
\begin{parts}
\part Determine el estimador de mínimo error cuadrático medio de $S$ a la vista de $x$.
\part Determine el error cuadrático medio condicionado a la observación, $\EE\{(S-\SMSE)^2|X=x\}$ para el estimador obtenido en el apartado a).
\part Determine el error cuadrático medio del estimador obtenido en el apartado a).
\end{parts}

\else

\question[20] % JCS

The joint distribution of random variables $S$ and $X$ is given by:
\[
p_{S,X}(s,x) = \sqrt{\frac{x(1-x)}{2\pi}} \exp\left(-\frac{(s-x(1-x))^2}{2x(1-x)} \right) , \qquad 0 < x < 1, \qquad s \in \mathbb{R} 
\]
\begin{parts}
\part Find the minimum mean square error (MSE) estimate of $S$ based on the observation $x$.
\part Compute the conditional MSE given observation $x$, $\EE\{(S-\SMSE)^2|X=x\}$, for the estimator obtained in part a).
\part Compute the MSE of the estimator obtained in part a).
\end{parts}

\fi


\begin{solution}
\begin{parts}
\part $\SMSE = X (1-X)$ 
\part $\EE\{(S-\SMSE)^2|X=x\} = x (1-x)$
\part $\EE\{(S-\SMSE)^2\} = \dfrac{1}{30}$
\end{parts}
\end{solution}
