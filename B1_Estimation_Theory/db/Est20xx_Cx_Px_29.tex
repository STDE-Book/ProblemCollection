\ifspanish

\question Se mide el par\'{a}metro determinista desconocido $s$, $s>0$, mediante dos sistemas, que proporcionan las observaciones
$$X_i=A_{i}s+N_i,	\quad	i=1,2$$

siendo $\left\{A_i\right\},\left\{N_i\right\}$, vv.aa. gaussianas e independientes entre s\'{\i}, con medias $\EE\left\{A_i\right\}=1$, $\EE\left\{N_i\right\}=0$, y varianzas $\left\{v_{Ai}\right\}$, $\left\{v_{Ni}\right\}$, respectivamente ($i=1,2$).
\begin{parts}
	\part Establ\'{e}zcase la expresi\'{o}n que proporciona el estimador ML de $s$, $\hat{S}_\text{ML}$.
	\part Calcule $\SML$ para el caso particular $v_{Ai}=0, i=1,2$.
	\part Calcule $\SML$ para el caso particular $v_{Ni}=0, i=1,2$.
\end{parts}

\begin{solution}   
\begin{parts}
\part $\SML =\argmin_s \left\{\ln\left[(s^{2}v_{A1}+v_{N1})(s^{2}v_{A2}+v_{N2})\right]
                       + \dfrac{(s-X_1)^2}{s^{2}v_{A1}+v_{N1}}
                       + \dfrac{(s-X_2)^2}{s^{2}v_{A2}+v_{N2}}\right\}$
\part $\SML = \dfrac{v_{N2}X_1 + v_{N1} X_2}{v_{N1} + v_{N2}}$
\part $\SML = \dfrac{1}{4}
              \sqrt{\left(\dfrac{X_1}{v_{A1}}+\dfrac{X_2}{v_{A2}}\right)^2 +
                    8 \left(\dfrac{X_1^2}{v_{A1}} + \dfrac{X_2^2}{v_{A2}}\right)}
              - \left(\dfrac{X_1}{v_{A1}} + \dfrac{X_2}{v_{A2}}\right)$
\end{parts}
\end{solution}

\question An unknown deterministic parameter $s$, $s>0$ is measured using two different systems, which provide observations
$$X_i=A_{i}s+N_i,	\quad	i=1,2$$

where $\left\{A_i\right\},\left\{N_i\right\}$, are independent Gaussian random vectors, with means $\EE\left\{A_i\right\}=1$, $\EE\left\{N_i\right\}=0$, and variances $\left\{v_{Ai}\right\}$, $\left\{v_{Ni}\right\}$, respectively ($i=1,2$).
\begin{parts}
	\part State the expression that defines the ML estimator of $s$, $\SML$.
	\part Obtain $\SML$ for the particular case $v_{Ai}=0, i=1,2$.
	\part Obtain $\SML$ for the particular case $v_{Ni}=0, i=1,2$.
\end{parts}

\begin{solution}   
\begin{parts}
\part $\SML =\argmin_s \left\{\ln\left[(s^2 v_{A1} + v_{N1})(s^2 v_{A2} + v_{N2})\right]
                              + \dfrac{(s-X_1)^2}{s^2 v_{A1} + v_{N1}}
                              + \dfrac{(s-X_2)^2}{s^2 v_{A2} + v_{N2}}
                       \right\}$
\part $\SML = \dfrac{v_{N2}X_1 + v_{N1} X_2}{v_{N1} + v_{N2}}$
\part $\SML = \dfrac{1}{4}
              \sqrt{\left(\dfrac{X_1}{v_{A1}}+\dfrac{X_2}{v_{A2}}\right)^2 +
                    8 \left(\dfrac{X_1^2}{v_{A1}} + \dfrac{X_2^2}{v_{A2}}\right)}
              - \left(\dfrac{X_1}{v_{A1}} + \dfrac{X_2}{v_{A2}}\right)$
\end{parts}
\end{solution}
