	\Leftrightarrow  & \quad \frac38 x^2 \dunodcero \eta \left( \frac34 - \frac{3}{16}x^2 \right)  \\
	\Leftrightarrow  & \quad \left(\frac38 + \frac{3}{16}\eta \right) x^2 \dunodcero  \frac34 \eta  \\
	\Leftrightarrow  & \quad x \dunodcero \frac{\frac34 \eta}{\frac38 + \frac{3}{16}\eta}            \\
	\Leftrightarrow  & \quad x \dunodcero \sqrt{\frac{4 \eta}{2 + \eta}} = \mu
\end{align*}
\part
