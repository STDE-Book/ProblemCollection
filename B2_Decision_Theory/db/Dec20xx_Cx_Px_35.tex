\question Considere el problema de decisión binaria dado por la observación ${\bf x} = (x_1,x_2) \in \mathbb{R}^2$ y verosimilitudes
\begin{align*}
p_{{\bf X}|H}({\bf x}|1) &= \exp(-x_1-x_2),  \qquad  x_1\ge 0, \quad x_2 \ge 0   \\
p_{{\bf X}|H}({\bf x}|0) &= 2, \qquad \qquad x_1\ge 0, \quad x_2\ge 0, \quad x_1+x_2 \le 1
\end{align*}
siendo $P_H(1)= 4/5$. 

\begin{parts}
\part Determine el decisor ML.
\part Determine el decisor MAP.
\part Determine la probabilidad de error del decisor ML.
\part Determine la probabilidad de falsa alarma del decisor MAP.
\end{parts}


\begin{solution}
\begin{parts}
\part $x_1+x_2 \dunodcero 1$
\part Decide $D=0$ si $\ln(2)<x_1+x_2<1$, decide $D=1$ en caso contrario.
\part $P_e = (1-2 e^{-1})/5$
\part $\pfa = \ln(2)^2$
\end{parts}
\end{solution}

\begin{parts}

\part Design the MAP classifier.
\part Calculate the probability of error of the ML classifier.
\part Calculate the probability of false alarm of the MAP classifier.
\end{parts}

\begin{solution}
\begin{parts}
\part $x_1+x_2 \dunodcero 1$
\part The MAP classifier decides $D=0$ if $\ln(2)<x_1+x_2<1$, and $D=1$ otherwise.
\part $P_e = (1-2 e^{-1})/5$
\part $\pfa = \ln(2)^2$
\end{parts}
\end{solution}
